        // 3. LOGIC APP

        // Initial Render
        function renderButtons(filter = "") {
            const container = document.getElementById('gridContainer');
            container.innerHTML = '';

            const filtered = dbTazkirah.filter(i => 
                i.title.toLowerCase().includes(filter.toLowerCase()) || 
                `hari ${i.day}`.includes(filter.toLowerCase())
            );

            filtered.forEach(item => {
                const btn = document.createElement('button');
                btn.className = 'day-btn animate-fade';
                btn.onclick = () => openReader(item.day);
                btn.innerHTML = `
                    <span class="day-label">HARI</span>
                    <span class="day-number">${item.day}</span>
                `;
                container.appendChild(btn);
            });
        }

        function filterDays() {
            const val = document.getElementById('searchInput').value;
            renderButtons(val);
        }

        // Reader
        const readerModal = document.getElementById('readerModal');
        const modalTitleDay = document.getElementById('modalTitleDay');
        const modalContent = document.getElementById('modalContent');
        const toast = document.getElementById('toast');

        function openReader(day) {
            const data = dbTazkirah.find(i => i.day === day);
            modalTitleDay.innerText = `HARI ${data.day < 10 ? '0' + data.day : data.day}`;
            
            modalContent.innerHTML = `
                <div class="mb-6 pb-6 border-b border-gray-100">
                    <h2 class="text-3xl font-black text-gray-900 leading-tight">${data.title}</h2>
                </div>
                <div>${data.body}</div>
            `;
            
            readerModal.classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        }

        function closeReader() {
            readerModal.classList.add('hidden');
            document.body.style.overflow = 'auto';
        }

        function copyContent() {
            const text = modalContent.innerText;
            navigator.clipboard.writeText(text).then(() => {
                toast.style.opacity = '1';
                toast.style.transform = 'translate(-50%, 0)';
                setTimeout(() => {
                    toast.style.opacity = '0';
                    toast.style.transform = 'translate(-50%, 40px)';
                }, 2000);
            });
        }

        // Start
        window.onload = () => renderButtons();

    </script>
